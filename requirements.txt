# =============================================================================
# requirements.txt — Project Dependencies
# =============================================================================
#
# What is this file?
#   This file lists every Python library your project needs to run.
#   When someone clones your repo, they run:
#       pip install -r requirements.txt
#   ...and Python installs everything automatically.
#
# Why pin versions (e.g., fastapi==0.115.0)?
#   Without pinning, pip installs the latest version — which may break your
#   code if the library's API changed. Pinning ensures the same versions
#   everywhere: your laptop, a teammate's laptop, Docker, and the server.
#
# NOTE: These versions are pinned for Python 3.13 compatibility.
#   Older versions (e.g. scikit-learn 1.5.x, numpy 1.x) do not ship
#   pre-built wheels for Python 3.13 and would fail to install without
#   a C/C++ compiler. Always match versions to your Python version.
# =============================================================================

# --- Web Framework ---
# FastAPI: the library we use to build our REST API (the web server that
# receives prediction requests and sends back results).
fastapi==0.115.0

# Uvicorn: the server that actually *runs* FastAPI. Think of FastAPI as the
# recipe and Uvicorn as the chef that executes it.
uvicorn==0.32.0

# Pydantic: handles data validation. When a request comes in, Pydantic checks
# that all required fields are present and have the right types before we
# even touch the model. Comes bundled with FastAPI.
pydantic==2.9.2

# --- Machine Learning ---
# scikit-learn: the most popular ML library in Python. Provides the
# RandomForestClassifier we use to predict churn, plus tools for
# preprocessing (scaling, encoding) and evaluation (accuracy, F1 score).
# 1.6.0+ is required for Python 3.13 pre-built wheels.
scikit-learn==1.6.1

# XGBoost: a powerful gradient boosting algorithm. Often outperforms
# Random Forest on tabular data. We train both and keep the better one.
# 2.1.0+ ships Python 3.13 wheels.
xgboost==2.1.4

# pandas: lets us load and manipulate our dataset as a "DataFrame" —
# think of it as a programmable Excel spreadsheet.
pandas==2.2.3

# numpy: the foundation of scientific Python. pandas and scikit-learn both
# use numpy arrays under the hood for fast math operations.
# numpy 2.x is required for Python 3.13.
numpy==2.2.0

# joblib: saves (serializes) our trained model to a .pkl file so we don't
# have to retrain every time the API starts up.
joblib==1.4.2

# --- Testing ---
# pytest: the standard Python testing framework. We write test functions and
# pytest finds and runs them automatically.
pytest==8.3.3

# httpx: an HTTP client that lets our tests send fake requests to the API
# without needing a real running server. FastAPI's TestClient uses it.
httpx==0.27.2

# --- Experiment Tracking ---
# MLflow: logs every training run automatically — parameters, metrics, and
# the model itself. Provides a web UI to compare runs side by side.
# This is what separates MLOps from plain ML: reproducibility + visibility.
# [skinny] = install without heavy optional deps (pyarrow, etc.) that
# don't build on Python 3.13 without a C compiler.
mlflow[skinny]==3.12.0
# pyarrow: let pip pick the best pre-built wheel for the current platform.
# Pinning an exact version risks choosing one with no wheel for the CI runner,
# forcing a source build that fails without a C compiler.
pyarrow>=14.0.0

# --- Utilities ---
# python-multipart: required by FastAPI to handle form data in requests.
python-multipart==0.0.12
